nom de la base de donnés : e-commerce
-------------------------------------------
CREATE TABLE categorie (
    id INT(11) PRIMARY KEY AUTO_INCREMENT,
    nom VARCHAR(255) COLLATE utf8mb4_unicode_ci NOT NULL,
    description TEXT COLLATE utf8mb4_unicode_ci,
    createur INT(11),
    date_creation DATETIME DEFAULT CURRENT_TIMESTAMP,
    date_modification DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

--------------------------------------------------
CREATE TABLE produit (
    id INT(11) PRIMARY KEY AUTO_INCREMENT,
    nom VARCHAR(255) COLLATE utf8mb4_unicode_ci NOT NULL,
    description TEXT COLLATE utf8mb4_unicode_ci,
    prix DECIMAL(10,2) NOT NULL,
    image VARCHAR(255) COLLATE utf8mb4_unicode_ci,
    categorie_id INT(11) NOT NULL,
    createur INT(11),
    date_creation DATETIME DEFAULT CURRENT_TIMESTAMP,
    date_modification DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    
    FOREIGN KEY (categorie_id) REFERENCES categorie(id),
    FOREIGN KEY (createur) REFERENCES visiteurs(id) -- Correction : référence à visiteurs
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;
-----------------------------------------------------------------------
CREATE TABLE visiteurs (
    id INT(11) PRIMARY KEY AUTO_INCREMENT,
    email VARCHAR(255) COLLATE utf8mb4_general_ci NOT NULL,
    mp VARCHAR(255) COLLATE utf8mb4_general_ci NOT NULL,
    nom VARCHAR(255) COLLATE utf8mb4_general_ci NOT NULL,
    prenom VARCHAR(255) COLLATE utf8mb4_general_ci NOT NULL,
    etat INT(11) NOT NULL DEFAULT 0,
    date_creation TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    e_modification DATE NOT NULL,
    telephone VARCHAR(12) COLLATE utf8mb4_general_ci NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci;
